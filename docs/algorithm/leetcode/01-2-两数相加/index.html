public class ListNode {
     public var val: Int
     public var next: ListNode?
     public init() { self.val = 0; self.next = nil; }
     public init(_ val: Int) { self.val = val; self.next = nil; }
     public init(_ val: Int, _ next: ListNode?) { self.val = val; self.next = next; }
}

class Solution {
    func addTwoNumbers(_ l1: ListNode?, _ l2: ListNode?) -> ListNode? {
        //å¦‚æœæŸä¸ªæ•°å­—ä¸º0ï¼Œé‚£ä¹ˆç›´æ¥è¿”å›å¦ä¸€ä¸ªæ•°å­—
        if l1?.val == 0 && l1?.next == nil { return l2 }
        if l2?.val == 0 && l2?.next == nil { return l1 }
        
        //åˆå§‹åŒ–
        let head: ListNode = ListNode()
        var tail: ListNode = head
        var carry = 0//è¿›ä½
        var p1 = l1
        var p2 = l2
        //éå†
        while(p1 != nil || p2 != nil || carry > 0) {
            let value1 = p1?.val ?? 0
            let value2 = p2?.val ?? 0
            let sum = value1 + value2 + carry
            let node = p1 ?? p2 ?? ListNode()//å¤ç”¨å·²æœ‰èŠ‚ç‚¹
            node.val = sum % 10//èµ‹å€¼
            tail.next = node //æ·»åŠ èŠ‚ç‚¹åˆ°å°¾éƒ¨
            tail = node //ç§»åŠ¨å°¾æŒ‡é’ˆ
                        
            carry = sum / 10 //è®¡ç®—è¿›ä½
            p1 = p1?.next //ç§»åŠ¨åˆ°ä¸‹ä¸€ä¸ª
            p2 = p2?.next //ç§»åŠ¨åˆ°ä¸‹ä¸€ä¸ª
        }
        return head.next
    }
}
