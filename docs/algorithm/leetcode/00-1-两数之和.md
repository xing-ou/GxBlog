class Solution {
   func twoSum(_ nums: [Int], _ target: Int) -> [Int] {
        let count = nums.count
        var final = [Int]()
        var map = [Int: Int].init(minimumCapacity: count) 
        let _ = nums.enumerated().first { (index, value) in
            let targetValue = target - value
            if let value = map[targetValue] {
                final = [value, index]
                return true
            } else {
                map[value] = index
                return false
            }
        }
        return final
    }
}
